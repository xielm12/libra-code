114  basisset_shell_struct(const basisset_shell_struct& ob){
115  type = ob.type;
116  contrCount = ob.contrCount;
117  shell_ID = ob.shell_ID;
118  exponentList = ob.exponentList;

120  }
121  ~basisset_shell_struct(){
122  exponentList.clear();
123  coeffList.clear();
124  }
125 
127  friend int operator !=(const basisset_shell_struct& m1, const basisset_shell_struct& m2){
128  if(m1.type != m2.type){ return 1; }
129  if(m1.contrCount != m2.contrCount){ return 1; }
130  if(m1.shell_ID != m2.shell_ID){ return 1; }
131  if(m1.exponentList.size() != m2.exponentList.size()){ return 1; }
132  if(m1.coeffList.size() != m2.coeffList.size()){ return 1; }
133 
134  int sz = m1.exponentList.size();
135  for(int i=0;i<sz;i++){
136  if(m1.exponentList[i] != m2.exponentList[i]) { return 1; }
137  if(m1.coeffList[i] != m2.coeffList[i]) { return 1; }
138  }
139  sz = m1.coeffList.size();
140  for(int i=0;i<sz;i++){
141  if(m1.coeffList[i] != m2.coeffList[i]) { return 1; }
142  }
143  return 0;
144  }
145 
147  friend int operator ==(const basisset_shell_struct& m1, const basisset_shell_struct& m2){
148  return !(m1!=m2);
149  }
