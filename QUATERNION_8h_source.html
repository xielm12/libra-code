63  //------------------ Basic quaternion operations ---------------------
64  QUATERNION inverse(){
65  QUATERNION tmp;
66  double norm=(Lt*Lt+Lx*Lx+Ly*Ly+Lz*Lz);
67  if(norm>0.0){
68  tmp.Lt = Lt/norm;
69  tmp.Lx = -Lx/norm;
70  tmp.Ly = -Ly/norm;
71  tmp.Lz = -Lz/norm;
72  }else{cout<<"Zero quaternion norm! Can not divide!"<<endl; }
73  return tmp;
74  }

143  QUATERNION tmp;
144  tmp.Lt = Lt*ob.Lt - (Lx*ob.Lx+Ly*ob.Ly+Lz*ob.Lz);
145  tmp.Lx = Lt*ob.Lx + ob.Lt*Lx + (Ly*ob.Lz-ob.Ly*Lz);
146  tmp.Ly = Lt*ob.Ly + ob.Lt*Ly + (Lz*ob.Lx-ob.Lz*Lx);
