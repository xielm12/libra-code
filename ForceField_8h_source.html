221  string get_atom_type(string elt,int geometry,string func_grp,int min_ring,int& coordination){
222  string res = "";
223  if(ForceField_Name == "UFF"){ res=uff_type(elt,geometry,func_grp,min_ring,coordination); }
224  else if(ForceField_Name == "DREIDING"){ res=dreiding_type(elt,geometry,func_grp,min_ring,coordination); }
225  else if(ForceField_Name == "GAFF"){ res=gaff_type(elt,geometry,func_grp,min_ring,coordination); }
226  else if(ForceField_Name == "MMFF94"){ res=mmff94_type(elt,geometry,func_grp,min_ring,coordination); }
227  else if(ForceField_Name == "TRIPOS"){ res=tripos_type(elt,geometry,func_grp,min_ring,coordination); }
228  else if(ForceField_Name == "TIP3P"){ res=tip3p_type(elt,geometry,func_grp,min_ring,coordination); }
229  return res;
