494  }// if

497  res = ( (st11==1)&&(st22==1) );
498  }// else
499 
500  }// if adjacent_vertices
501  }// else

503 
504  }// else

511

514  EDGE<EDGE_DATA> e; e = eg;

516  v1 = vertex_id_to_index(e.edge_vertex_id1);
517  v2 = vertex_id_to_index(e.edge_vertex_id2);
518 
519  if(v1==-1){ ADD_VERTEX(e.edge_vertex_id1); v1 = _V_-1;}
520  if(v2==-1){ ADD_VERTEX(e.edge_vertex_id2); v2 = _V_-1;}
521 
522

524  // Modify vertices

551  V[v2].adjacent_edges.push_back(_E_);

565 template <class VERTEX_DATA,class EDGE_DATA> int GRAPH<VERTEX_DATA,EDGE_DATA>::DELETE_EDGE(int e){
566 
567  int v1,v2;
568  if(e>=_E_){ std::cout<<"Error: Can not delete edge with index = "<<e<<" No such edge in a graph\n"; }
569  else{
570 
571  // Modify vertices
572  v1 = E[e].edge_vertex1;
573  v2 = E[e].edge_vertex2;
574  delete_item<int>(V[v1].adjacent_edges,e);
575  delete_item<int>(V[v2].adjacent_edges,e);
576  delete_item<int>(V[v1].adjacent_vertices,v2);
577  delete_item<int>(V[v2].adjacent_vertices,v1);
578  V[v1].vertex_degree--;
579  V[v2].vertex_degree--;
580 
581  // Modify edges
582  if(e==(_E_-1)){ ;; }
583  else{
584  int s1,s2;
585  s1 = E[_E_-1].edge_vertex1;
586  s2 = E[_E_-1].edge_vertex2;
587  E[e] = E[_E_-1];
588  E[e].edge_index = e;
589  replace_item<int>(V[s1].adjacent_edges,(_E_-1),e);
590  replace_item<int>(V[s2].adjacent_edges,(_E_-1),e);
591 
592  }
593 
594  // Modify graph
595  E.pop_back();
596  _E_--;
597  }// else
598 
599  return _E_;

603 template <class VERTEX_DATA,class EDGE_DATA> int GRAPH<VERTEX_DATA,EDGE_DATA>::DELETE_VERTEX(int v){
604 
605  if(v<_V_){
606  // Delete all edges iteratively (because of possible index mutations)
607  while(V[v].vertex_degree>0){
608  this->DELETE_EDGE(V[v].adjacent_edges[0]);
609  }
610  // Now delte the vertex (reindex the other vertices)
611  if(v==(_V_-1)){ ;; }
612  else{
613  // Modify vertices
614  V[v] = V[_V_-1];
615  V[v].vertex_index = v;
616 
617  // Modify edges
618  for(int i=0;i<V[v].vertex_degree;i++){
619  int e = V[v].adjacent_edges[i];
620  int v1,v2;
621  v1 = E[e].edge_vertex1;
622  v2 = E[e].edge_vertex2;
623  if(v1==(_V_-1)) { E[e].edge_vertex1 = v; replace_item<int>(V[v2].adjacent_vertices,(_V_-1),v); }
624  if(v2==(_V_-1)) { E[e].edge_vertex2 = v; replace_item<int>(V[v1].adjacent_vertices,(_V_-1),v); }
625  }
626  }
627 
628  // Modify graph
629  V.pop_back();
630  _V_--;
631 
632 
633  }else{
634  std::cout<<"Nothing to delete. No such vertex with index = "<<v<<"\n";
635  }
636  return _V_;
