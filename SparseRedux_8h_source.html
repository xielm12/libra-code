17 SparseMatrixBase<Derived>::sum() const
18 {
19  eigen_assert(rows()>0 && cols()>0 && "you are using a non initialized matrix");
20  Scalar res(0);
21  for (Index j=0; j<outerSize(); ++j)
22  for (typename Derived::InnerIterator iter(derived(),j); iter; ++iter)
23  res += iter.value();
24  return res;
